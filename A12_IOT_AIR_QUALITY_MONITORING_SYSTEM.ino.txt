#define buzzer 15

int a=0;

void setup()
{
  Serial.begin(9600);

  pinMode(gas,INPUT);

  pinMode(buzzer,OUTPUT);
  digitalWrite(buzzer,LOW);

  pinMode(2,OUTPUT);
  digitalWrite(2,LOW);

  pinMode(buzzer,OUTPUT);
  digitalWrite(buzzer,LOW);

  lcd.init();
  lcd.backlight();
  delay(100);

  lcd.clear();
  lcd.print("CONNECTING TO...");
  lcd.setCursor(0,1);
  lcd.print(ssid);
  delay(500);
  
  Serial.println("CONNECTING TO...");

  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, pass);
  
  while(WiFi.status() != WL_CONNECTED)
  {
    delay(100);
    Serial.println(".");
  }
  Serial.println(WiFi.localIP());

  Serial.println(" WIFI CONNECTED ");
  Serial.println(WiFi.localIP());
  delay(1000);

  Blynk.config(auth);

  ThingSpeak.begin(client);

  lcd.clear();
  lcd.print(" WIFI CONNECTED ");
  lcd.setCursor(0,1);
  lcd.print(WiFi.localIP());
  delay(1500);

  lcd.clear();
  lcd.print("   AIR QUALITY  ");
  lcd.setCursor(0,1);
  lcd.print("   MONITORING   ");

void senddata()
{
  ThingSpeak.setField(1,gas_sense);
  ThingSpeak.writeFields(ch_no, write_api);
}
void beep()
{
  digitalWrite(buzzer,HIGH);delay(500);digitalWrite(buzzer,LOW);delay(500);
  digitalWrite(buzzer,HIGH);delay(500);digitalWrite(buzzer,LOW);delay(500);
}
void loop()
{
  Blynk.run();
  
  b=b+1;
  if(b>40)
  {
    b=0;
  }
  
  gas_sense=analogRead(gas);
  gas_sense=map(gas_sense,0,4096,0,100);
  gas_sense=100-gas_sense;
  Serial.print("GAS : ");Serial.println(gas_sense);

  if(gas_sense <= 80)
  {
    gas_status="ALERT";
    a=a+1;
    beep();
  }
  else
  {
    a=0;
    gas_status="NORMAL";
  }
  
  if(a==3)
  {
    a=0;
    Serial.println("ALERT");
    Blynk.logEvent("gas_alert");
    Blynk.virtualWrite(V0,gas_sense);
  }

  if(b==40)
  {
    Blynk.virtualWrite(V0,gas_sense);
  }
  
  lcd.clear();
  lcd.print("GAS : ");lcd.print(gas_sense);lcd.print("%");
  lcd.setCursor(0,1);
  lcd.print("STATUS: ");lcd.print(gas_status);
  digitalWrite(2,HIGH);delay(300);digitalWrite(2,LOW);delay(200);
  
  senddata();
}
